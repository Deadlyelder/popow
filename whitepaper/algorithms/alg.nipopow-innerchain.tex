\begin{algorithm}[H]
    \caption{\label{alg.nipopow_construct_innerchain}constructInnerChain}
    \begin{algorithmic}[1]
        \Function{\sf constructInnerChain}{$\chain, i, boundary$}
            \Let{\textsf{innerChain}}{[\nobreakspace]}
            \Let{B}{\chain[\chain.length]}
            \Let{count}{0}
            \While{$count < boundary$}
                \Let{count}{count + 1}
                \Let{blockid}{B.\textsf{interlink}[i]}
                \Let{B}{\textsf{blockById}(blockid)}
                \State{\textsf{innerChain.append}($B$.interlink[$i$])}
            \EndWhile
            \State\Return{\textsf{innerChain}}
        \EndFunction
    \vskip8pt
    \end{algorithmic}
\end{algorithm}
