\begin{algorithm}[H]
    \caption{\label{alg.nipopow-verifier}The \textsf{Verify} algorithm
    for the NIPoPoW protocol}
    \begin{algorithmic}[1]
        \Function{\sf Verify$^Q_{m,k}$}{$\mathcal{P}$}
            \Let{(\tilde\Pi, \chi)}{\mathcal{P}[0]}
            \Let{\psi}{\emptyset}
            \For{$(\tilde{\Pi}, \chi) \in \mathcal{P}$}
                \If{$\mathsf{validProof}(\tilde\Pi, \chi)$}
                    \If{$\mathsf{maxChain}_m(\tilde\Pi, \psi)$}
                        \Let{\psi}{\chi}
                    \EndIf
                \EndIf
            \EndFor
            \State\Return{$Q(\epsilon^{|\chain| - k} \psi)$}
        \EndFunction
    \vskip8pt
    \end{algorithmic}
\end{algorithm}
