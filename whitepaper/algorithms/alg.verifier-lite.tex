\begin{algorithm}[H]
    \caption{\label{alg.nipopow-verifier}The \textsf{Verify} algorithm
    for the NiPoPoW protocol}
    \begin{algorithmic}[1]
        \Function{\sf Verify$_{m,k}$}{$\mathcal{P}$}
            \Let{best}{\mathcal{P}[0]}
            \For{$\tilde{\Pi} \in \mathcal{P}$}
                \Let{best}{\mathsf{maxChain}_m(best, \tilde\Pi)}
            \EndFor
            \State\Return{best}
        \EndFunction
    \vskip8pt
    \end{algorithmic}
\end{algorithm}
