\begin{algorithm}[H]
    \caption{\label{alg.verifier-full}The full verifier}
    \begin{algorithmic}[1]
        \Function{\sf Prove$_q$}{$\chain$}
            \State\Return{$\chain$}
        \EndFunction
        \Function{\sf Verify$_q$}{$\bar\chain$}
            \Let{\chain}{\mathsf{maxvalid}(\bar\chain)}
            \State\Return{$q(\chain)$}
        \EndFunction
    \vskip8pt
    \end{algorithmic}
\end{algorithm}
