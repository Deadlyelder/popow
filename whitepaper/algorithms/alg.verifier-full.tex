\begin{algorithm}[H]
    \caption{\label{alg.verifier_full}The full verifier}
    \begin{algorithmic}[1]
        \Function{\sf Prove}{$\chain$}
            \State\Return{$\chain$}
        \EndFunction
        \Function{\sf Verify}{$\bar\chain$}
            \Let{\chain}{\mathsf{maxvalid}(\bar\chain)}
            \State\Return{$\chain^{k \suffix}$}
        \EndFunction
    \vskip8pt
    \end{algorithmic}
\end{algorithm}
